


source_group(
    TREE "${PROJECT_SOURCE_DIR}/include"
    PREFIX "Header Files"
    FILES ${HEADER_LIST})
    
file(GLOB HEADER_LIST  
${INCLUDE_DIR}
${INCLUDE_DIR}/shared-memory)
    


file(GLOB SOURCE_LIST  ${CMAKE_SOURCE_DIR}/source/*.c)



find_package(PkgConfig REQUIRED)

pkg_check_modules(GLIB REQUIRED glib-2.0)
pkg_check_modules(GIO REQUIRED gio-2.0)
pkg_check_modules(GOBJECT REQUIRED gobject-2.0)

add_library(${PROJECT_NAME} STATIC ${SOURCE_LIST})

target_include_directories(${PROJECT_NAME}
    PUBLIC
        ${CMAKE_SOURCE_DIR}/include
        ${CMAKE_SOURCE_DIR}/include/shared-memory

        ${GLIB_INCLUDE_DIRS} 
        ${GIO_INCLUDE_DIRS} 
        ${GOBJECT_INCLUDE_DIRS}
)

target_compile_options(${PROJECT_NAME} PUBLIC ${GLIB_CFLAGS})
target_compile_options(${PROJECT_NAME} PUBLIC ${GIO_CFLAGS})
target_compile_options(${PROJECT_NAME} PUBLIC ${GOBJECT_CFLAGS})


target_link_libraries(${PROJECT_NAME} PUBLIC 
${GLIB_LIBRARY}  
${GIO_LIBRARY} 
${GOBJECT_LIBRARY})


target_link_libraries(${PROJECT_NAME} PUBLIC 
${GLIB_LINK_LIBRARY} 
${GIO_LINK_LIBRARY} 
${GOBJECT_LINK_LIBRARY}) 



#CACHE INTERNAL "${PROJECT_NAME}: Include Directories" FORCE)
