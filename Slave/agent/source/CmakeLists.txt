
find_package(PkgConfig REQUIRED)

pkg_check_modules(GLIB REQUIRED glib-2.0)
pkg_check_modules(GIO REQUIRED gio-2.0)
pkg_check_modules(GOBJECT REQUIRED gobject-2.0)
pkg_check_modules(JSON_GLIB REQUIRED json-glib-1.0)
pkg_check_modules(LIBSOUP REQUIRED libsoup-2.4)


file(GLOB SOURCE 
${CMAKE_SOURCE_DIR}/agent/source/*.c 
${CMAKE_SOURCE_DIR}/agent/source/file-transfer/*.c 
${CMAKE_SOURCE_DIR}/agent/source/agent-state/*.c
${CMAKE_SOURCE_DIR}/shared-items/*.c)


add_executable(agent ${SOURCE})


set_target_properties(agent PROPERTIES 
 RUNTIME_OUTPUT_DIRECTORY "${CMAKE_SOURCE_DIR}/bin")



target_include_directories(agent 
PUBLIC
    ${CMAKE_SOURCE_DIR}/agent/include
    ${CMAKE_SOURCE_DIR}/agent/include/file-transfer
    ${CMAKE_SOURCE_DIR}/agent/include/agent-state
	

    
    ${CMAKE_SOURCE_DIR}/shared-items
    ${LIBSOUP_INCLUDE_DIRS}
    ${GLIB_INCLUDE_DIRS} 
    ${GIO_INCLUDE_DIRS} 
    ${GOBJECT_INCLUDE_DIRS}
    ${JSON_GLIB_INCLUDE_DIRS}
    ${CONTROLLER_INCLUDE_DIRS})

target_compile_options(agent PUBLIC ${GLIB_CFLAGS})
target_compile_options(agent PUBLIC ${GIO_CFLAGS})
target_compile_options(agent PUBLIC ${GOBJECT_CFLAGS})
target_compile_options(agent PUBLIC ${JSON_GLIB_CFLAGS})
target_compile_options(agent PUBLIC ${CONTROLLER_CFLAGS})


target_link_directories(agent PUBLIC ${INCLUDE_LIBRARY} )  

target_link_libraries(agent PUBLIC 
    kernel32.lib
    user32.lib
    gdi32.lib
    winspool.lib
    shell32.lib
    ole32.lib
    oleaut32.lib
    uuid.lib
    comdlg32.lib
    advapi32.lib
    intl.lib
    gio-2.0.lib
    glib-2.0.lib
    gobject-2.0.lib
    gmodule-2.0.lib
    gstreamer-1.0.lib
    json-glib-1.0.lib
    soup-2.4.lib
    gstsdp-1.0.lib
    gstrtp-1.0.lib
    gstcontroller-1.0.lib
)
